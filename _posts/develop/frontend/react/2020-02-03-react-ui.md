# UI Framework

## Material-UI

### Install

```
$ npm install --save @material-ui/core

$npm install --save @material-ui/icons
```

### Start

```
import React from 'react';
import './App.css';
import Button from '@material-ui/core/Button';

function App() {
  return (
    <div className="App">
      <Button variant="contained" color="primary">
        Hello World
      </Button>
    </div>
  );
}

export default App;
```
